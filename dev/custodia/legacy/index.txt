    <script>
        (function() {
            'use strict';

            const BROKER_PHONE = "+56934545677"; 
            let state = JSON.parse(localStorage.getItem('psicoandino_state')) || {
                brokerName: "Corredora",
                leads: [],
                properties: []
            };

            const save = () => localStorage.setItem('psicoandino_state', JSON.stringify(state));

            const UI = {
                leads: document.getElementById('view-leads'),
                inventory: document.getElementById('view-inventory'),
                capture: document.getElementById('capture-panel'),
                pastebox: document.getElementById('pastebox'),
                brokerName: document.getElementById('broker-name'),
                brokerStatus: document.getElementById('broker-status')
            };

            // --- ENGINE: REGEX & KEY DECODER ---
            function processCapture(text) {
                if (text.startsWith('REF-')) {
                    return decodeKey(text);
                }

                const name = text.match(/(?:Nombre|Name|Cliente|Solicitante)\s*:\s*([^\n\r]+)/i);
                const phone = text.match(/(?:Teléfono|WhatsApp|Fono|Celular|Phone)\s*:\s*([+\d\s]+)/i);
                const slug = text.match(/(?:Propiedad|Ref|Slug|Código)\s*:\s*([^\n\r\s]+)/i);

                if (name && phone) {
                    const cleanPhone = phone[1].replace(/\s/g, '');
                    const newLead = {
                        id: Date.now(),
                        name: name[1].trim(),
                        phone: cleanPhone,
                        slug: slug ? slug[1].trim() : 'General',
                        status: 'pending',
                        tier: 'B2',
                        archived: false
                    };
                    state.leads.unshift(newLead);
                    save();
                    renderLeads();
                    closeCapture();
                } else {
                    document.getElementById('capture-error').innerText = "Captura Fallida: Datos Incompletos";
                }
            }

            function decodeKey(key) {
                try {
                    const raw = atob(key.replace('REF-', ''));
                    const [name, phone, slug, score, tier] = raw.split('|');
                    
                    const existing = state.leads.find(l => l.phone === phone);
                    if (existing) {
                        existing.status = 'qualified';
                        existing.tier = tier;
                        existing.slug = slug;
                    } else {
                        state.leads.unshift({
                            id: Date.now(),
                            name, phone, slug, status: 'qualified', tier, archived: false
                        });
                    }
                    save();
                    renderLeads();
                    closeCapture();
                } catch(e) {
                    document.getElementById('capture-error').innerText = "Código REF- Inválido";
                }
            }

            // --- RENDERERS ---
            function renderLeads() {
                const active = state.leads.filter(l => !l.archived);
                if (active.length === 0) {
                    UI.leads.innerHTML = `<div class="empty-state">Esperando captura...<br>(Copia un lead y pega aquí)</div>`;
                    return;
                }
                UI.leads.innerHTML = active.map(l => `
                    <div class="card ${l.status}" data-id="${l.id}">
                        <div class="card-header">
                            <div class="card-info">
                                <h3>${l.name} ${l.status === 'qualified' ? '✅' : ''}</h3>
                                <p>${l.phone} • ${l.slug}</p>
                            </div>
                            <span class="tier-badge">${l.tier}</span>
                        </div>
                        <div class="matrix-row">
                            <button class="matrix-btn" data-action="send" data-tmpl="A">Tmpl A</button>
                            <button class="matrix-btn" data-action="send" data-tmpl="B1">Tmpl B1</button>
                            <button class="matrix-btn" data-action="send" data-tmpl="B2">Tmpl B2</button>
                            <button class="archive-btn" data-action="archive">📂</button>
                        </div>
                    </div>
                `).join('');
            }

            function renderInventory() {
                const inv = state.properties;
                const formHTML = `
                    <div class="card" style="border-style: dashed;">
                        <h3 style="font-size:14px; margin-bottom:10px;">Nuevo Activo</h3>
                        <div class="manual-fields">
                            <input type="text" id="p-slug" placeholder="Slug (ej: depto-providencia)">
                            <input type="text" id="p-price" placeholder="Precio (ej: UF 4.500)">
                            <button class="btn btn-primary" id="btn-save-prop">Guardar Activo</button>
                        </div>
                    </div>
                `;
                UI.inventory.innerHTML = formHTML + inv.map((p, idx) => `
                    <div class="card">
                        <div class="card-header">
                            <div class="card-info">
                                <h3>${p.slug}</h3>
                                <p>${p.price}</p>
                            </div>
                            <button class="archive-btn" data-action="del-prop" data-idx="${idx}">🗑️</button>
                        </div>
                        <div class="prop-card-actions">
                            <button class="btn btn-secondary" data-action="copy-link" data-slug="${p.slug}">Copiar Link Cliente</button>
                        </div>
                    </div>
                `).join('');
            }

            // --- EVENT DELEGATION (NON-NEGOTIABLE) ---
            UI.leads.addEventListener('click', e => {
                const trigger = e.target.closest('[data-action]');
                if (!trigger) return;
                const card = trigger.closest('.card');
                const id = parseInt(card.dataset.id);
                const lead = state.leads.find(l => l.id === id);

                if (trigger.dataset.action === 'archive') {
                    lead.archived = true;
                    save(); renderLeads();
                } else if (trigger.dataset.action === 'send') {
                    const tmpl = trigger.dataset.tmpl;
                    const msg = `Hola ${lead.name}, contactamos de Boutique Assets por la propiedad ${lead.slug}...`;
                    window.open(`https://wa.me/${lead.phone}?text=${encodeURIComponent(msg)}`);
                }
            });

            UI.inventory.addEventListener('click', e => {
                const trigger = e.target.closest('[data-action], #btn-save-prop');
                if (!trigger) return;

                if (trigger.id === 'btn-save-prop') {
                    const s = document.getElementById('p-slug').value;
                    const p = document.getElementById('p-price').value;
                    if (s && p) {
                        state.properties.push({ slug: s, price: p });
                        save(); renderInventory();
                    }
                } else if (trigger.dataset.action === 'del-prop') {
                    state.properties.splice(trigger.dataset.idx, 1);
                    save(); renderInventory();
                } else if (trigger.dataset.action === 'copy-link') {
                    const link = `${window.location.origin}${window.location.pathname}?form=${trigger.dataset.slug}`;
                    navigator.clipboard.writeText(link);
                    trigger.innerText = "¡Copiado!";
                    setTimeout(() => trigger.innerText = "Copiar Link Cliente", 2000);
                }
            });

            // --- UI CONTROLS ---
            const openCapture = () => {
                UI.capture.classList.remove('hidden');
                UI.pastebox.focus();
            };
            const closeCapture = () => {
                UI.capture.classList.add('hidden');
                UI.pastebox.value = '';
                document.getElementById('capture-error').innerText = '';
            };

            document.getElementById('fab').addEventListener('click', openCapture);
            document.getElementById('btn-close-capture').addEventListener('click', closeCapture);

            document.getElementById('btn-toggle-mode').addEventListener('click', e => {
                const isManual = document.getElementById('paste-mode').classList.toggle('hidden');
                document.getElementById('manual-mode').classList.toggle('hidden');
                e.target.innerText = isManual ? "Modo Pegado" : "Carga Manual";
            });

            document.getElementById('btn-action-save').addEventListener('click', () => {
                if (!document.getElementById('manual-mode').classList.contains('hidden')) {
                    const n = document.getElementById('m-name').value;
                    const p = document.getElementById('m-phone').value;
                    const s = document.getElementById('m-slug').value;
                    if (n && p) {
                        processCapture(`Nombre: ${n}\nTeléfono: ${p}\nPropiedad: ${s || 'Manual'}`);
                    }
                } else {
                    processCapture(UI.pastebox.value);
                }
            });

            document.querySelectorAll('.nav-item').forEach(item => {
                item.addEventListener('click', () => {
                    document.querySelectorAll('.nav-item').forEach(i => i.classList.remove('active'));
                    item.classList.add('active');
                    const view = item.dataset.view;
                    UI.leads.classList.add('hidden');
                    UI.inventory.classList.add('hidden');
                    document.getElementById(view).classList.remove('hidden');
                    if (view === 'view-inventory') renderInventory();
                });
            });

            // --- BROKER FIELD ---
            UI.brokerName.innerText = state.brokerName;
            UI.brokerName.addEventListener('focus', () => {
                UI.brokerStatus.innerText = '✏️';
                UI.brokerStatus.style.color = 'var(--scale-a)';
            });
            UI.brokerName.addEventListener('blur', () => {
                state.brokerName = UI.brokerName.innerText;
                save();
                UI.brokerStatus.innerText = '✅';
                setTimeout(() => UI.brokerStatus.innerText = '👤', 1500);
            });

            // --- CLIENT LOGIC ---
            const params = new URLSearchParams(window.location.search);
            const clientSlug = params.get('form');
            if (clientSlug) {
                document.getElementById('admin-app-shell').classList.add('hidden');
                document.getElementById('client-form-view').classList.remove('hidden');
                document.getElementById('client-prop-title').innerText = clientSlug.toUpperCase();

                document.getElementById('q-form').addEventListener('submit', e => {
                    e.preventDefault();
                    const fd = new FormData(e.target);
                    let score = 0;
                    fd.forEach(v => score += parseInt(v));

                    let tier = 'B2';
                    if (score >= 40) tier = 'A';
                    else if (score >= 20) tier = 'B1';

                    const cName = document.getElementById('c-name').value || "Interesado";
                    const cPhone = document.getElementById('c-phone').value || "S/N";
                    
                    const payload = `${cName}|${cPhone}|${clientSlug}|${score}|${tier}`;
                    const key = `REF-${btoa(payload)}`;

                    const msg = `CAPTURA_PSICO | ${cName} | Prop: ${clientSlug} | Score: ${score} | Tier: ${tier}\n\nCódigo: ${key}`;
                    window.location.href = `https://wa.me/${BROKER_PHONE}?text=${encodeURIComponent(msg)}`;
                });
            }

            renderLeads();
        })();
    </script>
